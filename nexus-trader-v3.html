// server.js — npm install express cors node-fetch const express=require('express'),cors=require('cors'),fetch=require('node-fetch'); const app=express(); app.use(express.json());app.use(cors()); const SECRET=process.env.MT5_SECRET||'token'; const CLAUDE=process.env.ANTHROPIC_KEY||''; const signals=[]; app.post('/mt5-webhook',async(req,res)=>{ const d=req.body; if(d.token!==SECRET)return res.status(401).json({err:'Unauthorized'}); let aiSig='HOLD'; if(CLAUDE){ const r=await fetch('https://api.anthropic.com/v1/messages',{ method:'POST', headers:{'Content-Type':'application/json','x-api-key':CLAUDE,'anthropic-version':'2023-06-01'}, body:JSON.stringify({model:'claude-sonnet-4-6',max_tokens:10, messages:[{role:'user',content:`${d.symbol} RSI=${d.rsi} MA20=${d.ma20} close=${d.close}. BUY SELL or HOLD only.`}]}) }); const j=await r.json(); aiSig=(j.content?.[0]?.text||'HOLD').trim(); } signals.unshift({...d,aiSignal:aiSig,received:new Date().toISOString()}); if(signals.length>200)signals.pop(); res.json({ok:true,signal:aiSig}); }); app.get('/signals',(req,res)=>res.json(signals.slice(0,50))); app.get('/health',(req,res)=>res.json({ok:true})); app.listen(process.env.PORT||3000,()=>console.log('MT5 server running'));
